# OP CLI Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a CLI that wraps `@octothorpes/core` to provide all existing OP functionality from the command line.

**Architecture:** Modular `commander`-based CLI in a new standalone repo. Each command lives in its own file under `commands/`, shared config loading in `lib/client.js`, shared output formatting in `lib/format.js`. The CLI is a pure consumer of `createClient` from `octothorpes` -- no new business logic.

**Tech Stack:** Node.js (ESM), `commander`, `dotenv`, `octothorpes` (linked locally during dev via `file:` path)

**Design doc:** `docs/plans/2026-02-27-op-cli-design.md` in the octothorp.es repo

---

- The `octothorpes` package at `~/dev/octothorp.es/packages/core` must exist
- A SPARQL endpoint must be running for integration testing (e.g. `http://0.0.0.0:7878`)
- The OP dev server is NOT required -- the CLI talks directly to SPARQL

```javascript
import { createClient } from 'octothorpes'
import { readFileSync, existsSync } from 'fs'
import { resolve, join } from 'path'
import { homedir } from 'os'
import { config as loadDotenv } from 'dotenv'

const CONFIG_FILENAME = 'op.config.json'
const ENV_FILENAME = '.env'

/**
 * Resolves config file paths. Checks cwd first, then ~/.config/op/.
 * @param {string} cwd - Directory to check first
 * @returns {{ configPath: string|null, envPath: string|null }}
 */
function resolveConfigPaths(cwd) {
  const globalDir = join(homedir(), '.config', 'op')
  const candidates = [cwd, globalDir]

  let configPath = null
  let envPath = null

  for (const dir of candidates) {
    const cp = join(dir, CONFIG_FILENAME)
    if (!configPath && existsSync(cp)) configPath = cp
    const ep = join(dir, ENV_FILENAME)
    if (!envPath && existsSync(ep)) envPath = ep
  }

  return { configPath, envPath }
}

/**
 * Loads config from op.config.json and .env, creates an OP client.
 * @param {string} [cwd=process.cwd()] - Working directory to search for config
 * @returns {Promise<{ client: Object, config: Object }>}
 */
export async function loadConfig(cwd = process.cwd()) {
  const { configPath, envPath } = resolveConfigPaths(cwd)

  if (!configPath && !envPath) {
    throw new Error(
      'No OP config found. Run `op init` to create one, or check that .env and op.config.json exist in the current directory or ~/.config/op/'
    )
  }

  // Load .env if found
  if (envPath) {
    loadDotenv({ path: envPath })
  }

  // Load op.config.json if found
  let opConfig = {}
  if (configPath) {
    opConfig = JSON.parse(readFileSync(configPath, 'utf-8'))
  }

  const instance = opConfig.instance
  const endpoint = process.env.SPARQL_ENDPOINT
  const user = process.env.SPARQL_USER || ''
  const password = process.env.SPARQL_PASSWORD || ''

  if (!instance) {
    throw new Error('Missing "instance" in op.config.json')
  }
  if (!endpoint) {
    throw new Error('Missing SPARQL_ENDPOINT in .env')
  }

  const client = createClient({
    instance,
    sparql: { endpoint, user, password },
  })

  return { client, config: opConfig }
}
```

```javascript
const ANSI_COLORS = {
  red: '\x1b[31m',
  green: '\x1b[32m',
  yellow: '\x1b[33m',
  blue: '\x1b[34m',
  magenta: '\x1b[35m',
  cyan: '\x1b[36m',
  white: '\x1b[37m',
}
const RESET = '\x1b[0m'

/**
 * Wraps a string in ANSI color codes.
 * @param {string} text
 * @param {string} [color='magenta']
 * @returns {string}
 */
export function colorTerm(text, color = 'magenta') {
  const code = ANSI_COLORS[color] || ANSI_COLORS.magenta
  return `${code}${text}${RESET}`
}

function formatDate(timestamp) {
  if (!timestamp || isNaN(timestamp)) return ''
  return new Date(timestamp).toISOString().split('T')[0]
}

function resultCount(n) {
  return `${n} result${n === 1 ? '' : 's'}`
}

/**
 * Format blobject results as card blocks.
 * @param {Array} blobjects
 * @param {string} [termColor='magenta']
 * @returns {string}
 */
export function formatBlobjects(blobjects, termColor = 'magenta') {
  const cards = blobjects.map((b) => {
    const lines = []
    lines.push(`── ${b['@id']} ──`)
    if (b.title) lines.push(`Title:       ${b.title}`)
    if (b.description) lines.push(`Description: ${b.description}`)

    const terms = []
    const links = []
    if (b.octothorpes) {
      for (const o of b.octothorpes) {
        if (typeof o === 'string') terms.push(colorTerm(o, termColor))
        else if (o.uri) links.push(`${o.uri} (${o.type})`)
      }
    }
    if (terms.length) lines.push(`Terms:       ${terms.join(', ')}`)
    if (links.length) lines.push(`Links:       ${links.join(', ')}`)

    const date = formatDate(b.date || b.postDate)
    if (date) lines.push(`Indexed:     ${date}`)

    return lines.join('\n')
  })

  return cards.join('\n\n') + '\n\n' + resultCount(blobjects.length)
}

/**
 * Format page/link results as compact lines.
 * @param {Array} pages
 * @returns {string}
 */
export function formatPages(pages) {
  const lines = pages.map((p) => {
    const role = `[${p.role}]`.padEnd(10)
    const title = p.title ? `"${p.title}"` : ''
    const date = formatDate(p.date)
    return `${role}${p.uri}  ${title}  ${date}`.trimEnd()
  })

  return lines.join('\n') + '\n\n' + resultCount(pages.length)
}

/**
 * Format term results as colored term + date lines.
 * @param {Array} terms
 * @param {string} [termColor='magenta']
 * @returns {string}
 */
export function formatTerms(terms, termColor = 'magenta') {
  const lines = terms.map((t) => {
    const term = colorTerm(t.term, termColor).padEnd(30)
    const date = formatDate(t.date)
    return `${term}${date}`
  })

  return lines.join('\n') + '\n\n' + resultCount(terms.length)
}

/**
 * Format domain results as plain URI list.
 * @param {Array} domains
 * @returns {string}
 */
export function formatDomains(domains) {
  const lines = domains.map((d) => d.uri || d.d?.value || String(d))

  return lines.join('\n') + '\n\n' + resultCount(domains.length)
}

/**
 * Format raw SPARQL bindings as numbered key-value cards.
 * @param {Array} bindings
 * @returns {string}
 */
export function formatBindings(bindings) {
  const cards = bindings.map((b, i) => {
    const lines = [`── ${i + 1} ──`]
    for (const [key, val] of Object.entries(b)) {
      lines.push(`${key}:`.padEnd(10) + (val?.value ?? String(val)))
    }
    return lines.join('\n')
  })

  return cards.join('\n\n') + '\n\n' + resultCount(bindings.length)
}
```

```javascript
import { writeFileSync, existsSync, mkdirSync } from 'fs'
import { join } from 'path'
import { createInterface } from 'readline'

function ask(rl, question, defaultValue = '') {
  const prompt = defaultValue ? `${question} [${defaultValue}]: ` : `${question}: `
  return new Promise((resolve) => {
    rl.question(prompt, (answer) => {
      resolve(answer.trim() || defaultValue)
    })
  })
}

export async function init(options) {
  const dir = options.global
    ? join((await import('os')).homedir(), '.config', 'op')
    : process.cwd()

  const configPath = join(dir, 'op.config.json')
  const envPath = join(dir, '.env')

  if (existsSync(configPath) || existsSync(envPath)) {
    const rl = createInterface({ input: process.stdin, output: process.stdout })
    const overwrite = await ask(rl, 'Config files already exist. Overwrite? (y/N)', 'N')
    if (overwrite.toLowerCase() !== 'y') {
      rl.close()
      console.log('Cancelled.')
      return
    }
    rl.close()
  }

  const rl = createInterface({ input: process.stdin, output: process.stdout })

  console.log('\nOP CLI Setup\n')

  const instance = await ask(rl, 'OP instance URL (with trailing slash)', 'https://octothorp.es/')
  const endpoint = await ask(rl, 'SPARQL endpoint', 'http://0.0.0.0:7878')
  const user = await ask(rl, 'SPARQL username (leave blank if none)', '')
  const password = await ask(rl, 'SPARQL password (leave blank if none)', '')

  rl.close()

  // Ensure directory exists
  if (!existsSync(dir)) {
    mkdirSync(dir, { recursive: true })
  }

  // Write op.config.json
  const config = {
    instance,
    profile: {},
    termColor: 'magenta',
  }
  writeFileSync(configPath, JSON.stringify(config, null, 2) + '\n')
  console.log(`\nWrote ${configPath}`)

  // Write .env
  const envContent = [
    `SPARQL_ENDPOINT=${endpoint}`,
    `SPARQL_USER=${user}`,
    `SPARQL_PASSWORD=${password}`,
    '',
  ].join('\n')
  writeFileSync(envPath, envContent)
  console.log(`Wrote ${envPath}`)

  console.log('\nDone. Run `op get terms posted` to test your connection.')
}
```

```javascript
import { loadConfig } from '../lib/client.js'
import { formatBlobjects, formatPages, formatTerms, formatDomains } from '../lib/format.js'

// Map `what` values to their result format type
const RESULT_FORMATS = {
  everything: 'blobjects',
  blobjects: 'blobjects',
  whatever: 'blobjects',
  pages: 'pages',
  links: 'pages',
  mentions: 'pages',
  backlinks: 'pages',
  citations: 'pages',
  bookmarks: 'pages',
  thorpes: 'terms',
  octothorpes: 'terms',
  tags: 'terms',
  terms: 'terms',
  domains: 'domains',
}

export async function get(what, by, options) {
  const { client, config } = await loadConfig()
  const termColor = config.termColor || 'magenta'

  const queryOptions = {}
  if (options.s) queryOptions.s = options.s
  if (options.o) queryOptions.o = options.o
  if (options.notS) queryOptions['not-s'] = options.notS
  if (options.notO) queryOptions['not-o'] = options.notO
  if (options.match) queryOptions.match = options.match
  if (options.limit) queryOptions.limit = options.limit
  if (options.offset) queryOptions.offset = options.offset
  if (options.when) queryOptions.when = options.when

  const result = await client.get({ what, by, ...queryOptions })

  if (options.raw) {
    console.log(JSON.stringify(result, null, 2))
    return
  }

  const format = RESULT_FORMATS[what] || 'pages'
  switch (format) {
    case 'blobjects':
      console.log(formatBlobjects(result.results, termColor))
      break
    case 'terms':
      console.log(formatTerms(result.results, termColor))
      break
    case 'domains':
      console.log(formatDomains(result.results))
      break
    case 'pages':
    default:
      console.log(formatPages(result.results))
      break
  }
}
```

```javascript
import { loadConfig } from '../lib/client.js'
import { formatBindings } from '../lib/format.js'

const VALID_METHODS = ['terms', 'term', 'domains', 'domain', 'backlinks', 'bookmarks']

export async function fast(method, value, options) {
  if (!VALID_METHODS.includes(method)) {
    console.error(`Unknown fast method: ${method}`)
    console.error(`Valid methods: ${VALID_METHODS.join(', ')}`)
    process.exit(1)
  }

  const needsValue = ['term', 'domain']
  if (needsValue.includes(method) && !value) {
    console.error(`op fast ${method} requires a value argument`)
    process.exit(1)
  }

  const { client } = await loadConfig()

  const result = needsValue.includes(method)
    ? await client.getfast[method](value)
    : await client.getfast[method]()

  if (options.raw) {
    console.log(JSON.stringify(result, null, 2))
    return
  }

  // fast.term() and fast.domain() return { pages, bookmarks } or { backlinks, bookmarks }
  if (typeof result === 'object' && !Array.isArray(result)) {
    for (const [key, bindings] of Object.entries(result)) {
      if (Array.isArray(bindings) && bindings.length > 0) {
        console.log(`\n${key}:\n`)
        console.log(formatBindings(bindings))
      } else if (Array.isArray(bindings)) {
        console.log(`\n${key}: (none)`)
      }
    }
    return
  }

  // Everything else returns a flat bindings array
  console.log(formatBindings(result))
}
```

```javascript
import { loadConfig } from '../lib/client.js'
import { formatBlobjects } from '../lib/format.js'

export async function harmonize(url, options) {
  const { client, config } = await loadConfig()
  const termColor = config.termColor || 'magenta'
  const harmonizerName = options.harmonizer || 'default'

  try {
    const response = await fetch(url)
    if (!response.ok) {
      console.error(`Failed to fetch ${url}: ${response.status} ${response.statusText}`)
      process.exit(1)
    }
    const html = await response.text()
    const blobject = await client.harmonize(html, harmonizerName)

    if (options.raw) {
      console.log(JSON.stringify(blobject, null, 2))
      return
    }

    console.log(formatBlobjects([blobject], termColor))
  } catch (err) {
    console.error(`Failed to harmonize ${url}: ${err.message}`)
    process.exit(1)
  }
}
```

```javascript
import { loadConfig } from '../lib/client.js'

/**
 * Builds the SPARQL INSERT query for domain registration.
 * Exported for testing.
 */
export function buildRegisterQuery(domain) {
  const clean = domain.replace(/\/$/, '')
  return `<${clean}> octo:verified 'true' .\n<${clean}> rdf:type <octo:Origin> .`
}

export async function register(domain, options) {
  const { client } = await loadConfig()

  try {
    const query = buildRegisterQuery(domain)
    await client.sparql.insert(query)
    console.log(`Registered: ${domain.replace(/\/$/, '')}`)
  } catch (err) {
    console.error(`Failed to register ${domain}: ${err.message}`)
    process.exit(1)
  }
}
```
